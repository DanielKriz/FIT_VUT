19  void rotateImage(const cv::Mat &input, cv::Mat &output, double alpha, double beta, double gamma, double dx, double dy, double dz, double f)
20  {
21  alpha = (alpha - 90.) * CV_PI/180.;
22  beta = (beta - 90.) * CV_PI/180.;
23  gamma = (gamma - 90.) * CV_PI/180.;
24 
25  // get width and height for ease of use in matrices
26  double w = (double)input.cols;
27  double h = (double)input.rows;
28 
29  // Projection 2D -> 3D matrix
30  cv::Mat A1 = (cv::Mat_<double>(4,3) <<
31  1, 0, -w/2,
32  0, 1, -h/2,
33  0, 0, 0,
34  0, 0, 1);
35 
36  // Rotation matrices around the X, Y, and Z axis
37  cv::Mat RX = (cv::Mat_<double>(4, 4) <<
38  1, 0, 0, 0,
39  0, cos(alpha), -sin(alpha), 0,
40  0, sin(alpha), cos(alpha), 0,
41  0, 0, 0, 1);
42 
43  cv::Mat RY = (cv::Mat_<double>(4, 4) <<
44  cos(beta), 0, -sin(beta), 0,
45  0, 1, 0, 0,
46  sin(beta), 0, cos(beta), 0,
47  0, 0, 0, 1);
48 
49  cv::Mat RZ = (cv::Mat_<double>(4, 4) <<
50  cos(gamma), -sin(gamma), 0, 0,
51  sin(gamma), cos(gamma), 0, 0,
52  0, 0, 1, 0,
53  0, 0, 0, 1);
54 
55  // Composed rotation matrix with (RX, RY, RZ)
56  cv::Mat R = RX * RY * RZ;
57 
58  // Translation matrix
59  cv::Mat T = (cv::Mat_<double>(4, 4) <<
60  1, 0, 0, dx,
61  0, 1, 0, dy,
62  0, 0, 1, dz,
63  0, 0, 0, 1);
64 
65  // 3D -> 2D matrix
66  cv::Mat A2 = (cv::Mat_<double>(3,4) <<
67  f, 0, w/2, 0,
68  0, f, h/2, 0,
69  0, 0, 1, 0);
70 
71  // Final transformation matrix
72  cv::Mat trans = A2 * (T * (R * A1));
73 
74  // Apply matrix transformation
75  cv::warpPerspective(input, output, trans, input.size(), cv::INTER_LANCZOS4);
76  }

297  uchar newHueBlue = hueDistBlue( rng );

303  cv::cvtColor(img, hsv, CV_BGR2HSV);

305  // For each pixel
306

308  cv::MatIterator_<cv::Vec3b> end = hsv.end<cv::Vec3b>();
309 
310  std::for_each(begin, end, [&](cv::Vec3b& p)
311  {
312  // H = 0; S = 1; V = 2

338  {
339  std::vector<cv::Mat> channels;

347  // TODO: Its BGR, not RGB - so maybe change indexes
348  // 0,1,2 -> 2,1,0
349

351  // Y = 0.299 * R

353  // U = 0.587 * G
