163 ///////////////////////////////////////////////////////////////////////////////////////////////
164 void DatasetGeneratorTransparent_t::opAddGradient(cv::Mat& m)
165 {
166 # ifdef GRADIENT
167  m_prngProbability = m_probability(m_rng);
168 
169  if ( m_prngProbability <= 25 ) // 25%
170  {
171  double alpha = .7;
172  double beta = 10;
173  double deltaAlpha = 1.3 / m.rows;
174  double deltaBeta = 80 / m.rows;
175  int dir = ( m_probability(m_rng) < 50 ) ? m_distGradient(m_rng) : - m_distGradient(m_rng);
176 
177  cv::Mat tmp = cv::Mat::zeros(cv::Size(2*m.cols, 2*m.rows), m.type());
178  int x_coord = m.cols / 2;
179  int y_coord = m.rows / 2;
180 
181  m.copyTo(tmp(cv::Rect(x_coord, y_coord, m.cols, m.rows)));
182 
183  ImageProcessing::rotateImage(tmp, tmp, 90, 90, 90 + dir, 0, 0, 200, 200);
184 
185  cv::Mat newImage;
186  tmp.copyTo(newImage);
187 
188  // Do the operation for each pixel: newImage(i,j) = alpha*img(i,j) + beta
189  for( int y = 0; y < tmp.rows; y++ )
190  {
191  for( int x = 0; x < tmp.cols; x++ )
192  {
193  for( int c = 0; c < 3; c++ )
194  {
195  newImage.at<cv::Vec3b>(y, x)[c] = cv::saturate_cast<uchar>( alpha * ( tmp.at<cv::Vec3b>(y, x)[c] ) + beta );
196  }
197  }
198 
199  if ( y > y_coord && y < (y_coord + m.cols) )
200  {
201  alpha += deltaAlpha;
202  beta += deltaBeta;
203  }
204  }
205 
206  ImageProcessing::rotateImage(newImage, newImage, 90, 90, 90 - dir, 0, 0, 200, 200);
207 
208  m = newImage(cv::Rect(x_coord, y_coord, m.cols, m.rows));
209  }
210 # endif
211 }
